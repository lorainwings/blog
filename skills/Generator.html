function run(gen) {
  return new Promise(function(resolve, reject) {
    if (typeof gen == 'function') gen = gen();

    // å¦‚æœ gen ä¸æ˜¯ä¸€ä¸ªè¿­ä»£å™¨

        return reject(e);
